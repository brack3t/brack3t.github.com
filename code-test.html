class Unit(UnitRent, UnitPromo, UnitDeposit, UnitSquareFeet, DateTimeAuditModel,
    LumberjackModel):

    community = models.ForeignKey(Community, related_name="units")
    floorplan = models.ForeignKey(Floorplan, related_name="units")
    address = models.CharField(max_length=255)
    number = models.CharField(max_length=20)
    available_date = models.DateField(blank=True, null=True)
    leases = models.ManyToManyField(Lease, blank=True, null=True,
        related_name="units")
    building = models.CharField(blank=True, default="", max_length=255)
    floor = models.CharField(blank=True, default="", max_length=255)
    location = models.CharField(blank=True, default="", max_length=255,
        help_text="Describe the location of the unit.")
    visible = models.BooleanField(default=True)

    class Meta:
        ordering = ("floorplan__name", "number")
        unique_together = ("address", "number", "community")

    def __unicode__(self):
        return u"%s - %s" % (self.number, self.floorplan.name)

    def save(self, *args, **kwargs):
        """
        Make sure cleans are called.
        """
        self.clean()
        super(Unit, self).save(*args, **kwargs)

    @property
    def rent(self):
        if self.rent_inherits:
            return self.floorplan.rent

        return super(Unit, self).rent

    @property
    def deposit(self):
        if self.deposit_inherits:
            return self.floorplan.deposit

        return super(Unit, self).deposit

    @property
    def promo(self):
        if self.promo_inherits:
            return self.floorplan.promo

        return super(Unit, self).promo

    @property
    def sq_ft(self):
        if self.sq_ft_inherits:
            return self.floorplan.sq_ft

        return super(Unit, self).sq_ft
